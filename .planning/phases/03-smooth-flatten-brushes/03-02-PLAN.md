Algorithm: For each vertex within the brush radius, compute a Gaussian-weighted average of all nearby offsets (within the same radius), then lerp the vertex's current offset toward that average. The lerp strength is `smooth_strength * gaussian_weight_of_this_vertex` so the center of the brush changes most.

```gdscript
# Smooth brush: nudge height offsets toward a Gaussian-weighted local average.
# Continuous — caller fires every frame while LMB held.
# strength: how strongly to pull toward average per call (0.0-1.0 range recommended, caller passes delta-scaled value).
# Returns dirty Array[Vector2i].
func apply_smooth_brush(world_pos: Vector3, radius: float, strength: float) -> Array:
    var dirty: Array = []
    if radius <= 0.0 or absf(strength) < 0.0001:
        return dirty

    # Step 1: Gather world-space vertex positions and their current offsets
    # within the brush radius across all relevant chunks.
    # We need cross-chunk sampling to avoid edge discontinuities in the average.
    var half_cs: float = chunk_size * 0.5
    var min_coord: Vector2i = get_chunk_coords(Vector3(world_pos.x - radius - half_cs, 0.0, world_pos.z - radius - half_cs))
    var max_coord: Vector2i = get_chunk_coords(Vector3(world_pos.x + radius + half_cs, 0.0, world_pos.z + radius + half_cs))

    # Collect all vertices in range: {wx, wz, current_offset}
    # We sample from ALL chunks in range (including those outside brush center chunk)
    # so the weighted average is global, not per-chunk.
    var sigma: float = radius * 0.5

    # Pass 1: Compute the Gaussian-weighted average offset across all in-range vertices.
    var weighted_sum: float = 0.0
    var weight_total: float = 0.0
    for cz in range(min_coord.y, max_coord.y + 1):
        for cx in range(min_coord.x, max_coord.x + 1):
            var coord: Vector2i = Vector2i(cx, cz)
            if not chunks.has(coord):
                continue
            var chunk = chunks[coord]
            if chunk == null or not "height_offsets" in chunk:
                continue
            if chunk.height_offsets.is_empty():
                chunk._ensure_offsets_sized()
            var res: int = chunk.resolution
            var cs: float = chunk.chunk_size
            var half: float = cs / 2.0
            for z in range(res + 1):
                for x in range(res + 1):
                    var local_x: float = (float(x) / float(res)) * cs - half
                    var local_z: float = (float(z) / float(res)) * cs - half
                    var wx: float = chunk.position.x + local_x
                    var wz: float = chunk.position.z + local_z
                    var dx: float = wx - world_pos.x
                    var dz: float = wz - world_pos.z
                    var dist_sq: float = dx * dx + dz * dz
                    if dist_sq > radius * radius:
                        continue
                    var w: float = exp(-dist_sq / (2.0 * sigma * sigma))
                    weighted_sum += chunk.height_offsets[z * (res + 1) + x] * w
                    weight_total += w

    if weight_total < 0.0001:
        return dirty

    var avg_offset: float = weighted_sum / weight_total

    # Pass 2: Nudge each in-range vertex toward avg_offset, weighted by Gaussian.
    for cz in range(min_coord.y, max_coord.y + 1):
        for cx in range(min_coord.x, max_coord.x + 1):
            var coord: Vector2i = Vector2i(cx, cz)
            if not chunks.has(coord):
                continue
            var chunk = chunks[coord]
            if chunk == null or not "height_offsets" in chunk:
                continue
            if chunk.height_offsets.is_empty():
                chunk._ensure_offsets_sized()
            var res: int = chunk.resolution
            var stride: int = res + 1
            var cs: float = chunk.chunk_size
            var half: float = cs / 2.0
            var touched: bool = false
            for z in range(res + 1):
                for x in range(res + 1):
                    var local_x: float = (float(x) / float(res)) * cs - half
                    var local_z: float = (float(z) / float(res)) * cs - half
                    var wx: float = chunk.position.x + local_x
                    var wz: float = chunk.position.z + local_z
                    var dx: float = wx - world_pos.x
                    var dz: float = wz - world_pos.z
                    var dist_sq: float = dx * dx + dz * dz
                    if dist_sq > radius * radius:
                        continue
                    var w: float = exp(-dist_sq / (2.0 * sigma * sigma))
                    var idx: int = z * stride + x
                    chunk.height_offsets[idx] = lerp(chunk.height_offsets[idx], avg_offset, strength * w)
                    touched = true
            if touched and not dirty.has(coord):
                dirty.append(coord)

    return dirty
```

Key design notes:
- Two passes are required: first gather the global average, then apply the nudge. A single-pass approach would read modified values mid-iteration, making results order-dependent and non-commutative.
- `strength` is passed as a small per-frame delta (caller uses same BRUSH_STRENGTH_PER_SEC * delta pattern but treats it as a lerp weight, so scale it accordingly — see Task 2).
- The cross-chunk global average prevents seam artifacts at chunk boundaries.

```gdscript
func _tick_brush(delta: float) -> void:
    if not is_editor_active:
        return
    if terrain_manager == null:
        return
    # Skip brush while camera-drag modifiers are held (LMB is used for orbit/pan)
    if Input.is_key_pressed(KEY_SHIFT) or Input.is_key_pressed(KEY_CTRL):
        return
    var lmb: bool = Input.is_mouse_button_pressed(MOUSE_BUTTON_LEFT)
    var rmb: bool = Input.is_mouse_button_pressed(MOUSE_BUTTON_RIGHT)
    if not lmb and not rmb:
        return
    var pos_v = _get_terrain_cursor_world_pos()
    if pos_v == null:
        return
    var world_pos: Vector3 = pos_v

    match brush_mode:
        BrushMode.RAISE_LOWER:
            var direction: float = 1.0 if lmb else -1.0
            var boost: float = BRUSH_BOOST_MULT if Input.is_key_pressed(KEY_R) else 1.0
            var strength: float = direction * BRUSH_STRENGTH_PER_SEC * boost * delta
            var dirty: Array = terrain_manager.apply_height_brush(world_pos, brush_radius, strength, brush_falloff)
            if dirty.size() > 0:
                _rebuild_dirty_chunks(dirty)
        BrushMode.SMOOTH:
            # Only fires on LMB — smooth has no directional opposite
            if not lmb:
                return
            # strength is a lerp weight per frame; 3.0 * delta gives ~0.05 at 60fps
            # which is a gentle nudge. Boost doubles the convergence rate.

            var strength: float = 3.0 * boost * delta
            var dirty: Array = terrain_manager.apply_smooth_brush(world_pos, brush_radius, strength)
            if dirty.size() > 0:
                _rebuild_dirty_chunks(dirty)
        BrushMode.FLATTEN:
            # Flatten logic added in Plan 03.
            # Stub: do nothing yet so mode cycling doesn't break anything.
            pass
```

The smooth strength constant `3.0` was chosen so that `3.0 * (1/60) ≈ 0.05` — a 5% lerp per frame toward the average, which feels responsive without being instant. Boost doubles it to ~10% per frame. These values can be tuned by the designer via the brush boost (R key).

Do NOT add a new constant for smooth strength — it's expressed inline as `3.0 * boost * delta` so it's easy to find and change.
